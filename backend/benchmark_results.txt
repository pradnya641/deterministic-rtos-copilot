 * UART0_Init - initialises UART0 at 9600 baud
 * (Assuming PCLK = 15 MHz)
 */
void UART0_Init(void)
{
    /* Step 1: Configure P0.0 (TXD0) and P0.1 (RXD0) 
       Clear bits [3:0] then set bits 3:0 to 0101 (value 5) */
    PINSEL0 &= ~(0xF << 0);
    PINSEL0 |=  (0x5 << 0);

    /* Step 2: Enable DLAB (Divisor Latch Access Bit) */
    U0LCR = 0x83; 

    /* Step 3: Set Baud Rate (9600)
       DLL = PCLK / (16 * baud) = 15M / (16 * 9600) = 97 (0x61) */
    U0DLL = 0x61;
    U0DLM = 0x00;

    /* Step 4: Disable DLAB, set 8-bit, no parity, 1 stop bit (8N1) */
    U0LCR = 0x03;

    /* Step 5: Enable FIFO and reset RX/TX buffers */
    U0FCR = 0x07;
}

/**
 * UART0_SendChar - Sends a single character (polling)
 */
void UART0_SendChar(char c)
{
    /* Wait until Transmit Holding Register (THR) is empty (bit 5 of LSR) */

    U0THR = c;
}

 * UART0_ReceiveChar - Receives a single character (polling)
 */
char UART0_ReceiveChar(void)
{
    /* Wait for Receiver Data Ready (RDR) (bit 0 of LSR) */
    while (!(U0LSR & (1 << 0)));
    return U0RBR;
}
